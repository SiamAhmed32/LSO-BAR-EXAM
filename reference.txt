======================== Dynamic ID Page Example

const SingleItemPage = async ({ params }: { params: { id: string } }) => {
  // console.log('Dynamic params:', params);

  const { id } = params;

  const itemData = await getById({ path: 'items', id });

  return (
    <Layout>
      <SingleItemComponent itemData={itemData} />
    </Layout>
  );
};

export default SingleItemPage;
